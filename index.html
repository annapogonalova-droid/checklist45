        function calculateResult() {

            const errors = document.querySelectorAll('.error:checked');

            const totalErrors = 45;

            const checkedCount = errors.length;

            const percent = Math.round((checkedCount / totalErrors) * 100);

            let advice = '';

            if (percent > 70) {

                advice = 'Критично! Много ошибок — срочно перестройте систему сопровождения, чтобы избежать оттока клиентов.';

            } else if (percent > 30) {

                advice = 'Есть проблемы. Оптимизируйте ключевые этапы, начните с онбординга и коммуникации.';

            } else {

                advice = 'Хорошо! Ваш процесс на хорошем уровне, но проверьте слабые места для идеала.';

            }

            document.getElementById('result').innerHTML = `

                <strong>Отмечено ошибок: ${checkedCount} из ${totalErrors} (${percent}%)</strong><br>

                <p>${advice}</p>

            `;
